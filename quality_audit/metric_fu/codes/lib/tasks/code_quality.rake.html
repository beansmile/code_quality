123    desc "rubocop - audit coding style"
  
124    # e.g.: rake code_quality:quality_audit:rubocop max_offenses=100
  
125    # options:
  
126    #   config_formula: use which formula for config, supports "github, "rails" or path_to_your_local_config.yml, default is "github"
  
127    #   cli_options: pass extract options, e.g.: cli_options="--show-cops"
  
128    #   max_offenses: if config max_offenses then audit it with detected offenses number in report, e.g.: max_offenses=100
  
129    task :rubocop => :prepare do |task|
  
130      run_audit task, "rubocop - RuboCop is a Ruby static code analyzer. Out of the box it will enforce many of the guidelines outlined in the community Ruby Style Guide." do
  
131        options = options_from_env(:config_formula, :cli_options, :max_offenses)
  
132
  
133        config_formulas = {
  
134          'github' => 'https://github.com/github/rubocop-github',
  
135          'rails' => 'https://github.com/rails/rails/blob/master/.rubocop.yml'
  
136        }
  
137
  
138        # prepare cli options
  
139        config_formula = options.fetch(:config_formula, 'github')
  
140        if config_formula && File.exists?(config_formula)
  
141          config_file = config_formula
  
142          puts "Using config file: #{config_file}"
  
143        else
  
144          gem_config_dir = File.expand_path("../../../config", __FILE__)
  
145          config_file    = "#{gem_config_dir}/rubocop-#{config_formula}.yml"
  
146          puts "Using config formula: [#{config_formula}](#{config_formulas[config_formula]})"
  
147        end
  
148        @report_path = report_path = "#{report_dir}/rubocop-report.html"
  
149
  
150        # generate report
  
151        report = `rubocop -c #{config_file} -S -R -P #{options[:cli_options]} --format offenses --format html -o #{report_path}`
  
152        puts report
  
153        puts "Report generated to #{report_path}"
  
154        show_in_browser File.realpath(report_path)
  
155
  
156        # if config max_offenses then audit it with detected offenses number in report
  
157        if options[:max_offenses]
  
158          if report[-20..-1] =~ /(\d+) *Total/
  
159            detected_offenses = $1.to_i
  
160            max_offenses = options[:max_offenses].to_i
  
161            audit_faild "Detected offenses #{colorize(detected_offenses, :yellow)} is more then #{colorize(max_offenses, :yellow)}, must improve your code quality or set a lower #{colorize("max_offenses", :black, :white)}" if detected_offenses > max_offenses
  
162          end
  
163        end
  
164      end
  
165    end

167    desc "metric_fu - many kinds of metrics"
  
168    # e.g.: rake code_quality:quality_audit:metric_fu metrics=stats,rails_best_practices,roodi rails_best_practices_max_offenses=9 roodi_max_offenses=10
  
169    # options:
  
170    #   metrics: default to run all metrics, can be config as: cane,churn,flay,flog,hotspots,rails_best_practices,rcov,reek,roodi,saikuro,stats
  
171    #   flay_max_offenses: offenses number for audit
  
172    #   cane_max_offenses: offenses number for audit
  
173    #   rails_best_practices_max_offenses: offenses number for audit
  
174    #   reek_max_offenses: offenses number for audit
  
175    #   roodi_max_offenses: offenses number for audit
  
176    task :metric_fu => :prepare do |task|
  
177      metrics_offenses_patterns = {
  
178        "flay" => /Total Score (\d+)/,
  
179        "cane" => /Total Violations (\d+)/,
  
180        "rails_best_practices" => /Found (\d+) errors/,
  
181        "reek" => /Found (\d+) code smells/,
  
182        "roodi" => /Found (\d+) errors/,
  
183      }
  
184      metrics_have_offenses = metrics_offenses_patterns.keys.map { |metric| "#{metric}_max_offenses".to_sym }
  
185      options = options_from_env(:metrics, *metrics_have_offenses)
  
186      run_audit task, "metric_fu - Code metrics from Flog, Flay, Saikuro, Churn, Reek, Roodi, Code Statistics, and Rails Best Practices. (and optionally RCov)" do
  
187        report_path = "#{report_dir}/metric_fu"
  
188        available_metrics = %w{cane churn flay flog hotspots rails_best_practices rcov reek roodi saikuro stats}
  
189        metric_fu_opts = ""
  
190        selected_metrics = available_metrics
  
191        if options[:metrics]
  
192          selected_metrics = options[:metrics].split(",")
  
193          disable_metrics = available_metrics - selected_metrics
  
194          selected_metrics_opt = selected_metrics.map { |m| "--#{m}" }.join(" ")
  
195          disable_metrics_opt = disable_metrics.map { |m| "--no-#{m}" }.join(" ")
  
196          metric_fu_opts = "#{selected_metrics_opt} #{disable_metrics_opt}"
  
197          puts "for metrics: #{selected_metrics.join(",")}"
  
198        end
  
199        # geneate report
  
200        report = `bundle exec metric_fu --no-open #{metric_fu_opts}`
  
201        FileUtils.remove_dir(report_path) if Dir.exists? report_path
  
202        FileUtils.mv("tmp/metric_fu/output", report_path, force: true)
  
203        puts report
  
204        puts "Report generated to #{report_path}"
  
205        show_in_browser File.realpath(report_path)
  
206        @report_path = "#{report_path}/index.html"
  
207
  
208        # audit report result
  
209        report_result_path = "tmp/metric_fu/report.yml"
  
210        if File.exists? report_result_path
  
211          require 'yaml'
  
212          report_result = YAML.load_file(report_result_path)
  
213          # if config #{metric}_max_offenses then audit it with report result
  
214          audit_failures = []
  
215          metrics_offenses_patterns.each do |metric, pattern|
  
216            option_key = "#{metric}_max_offenses".to_sym
  
217            if options[option_key]
  
218              detected_offenses = report_result[metric.to_sym][:total].to_s.match(pattern)[1].to_i rescue 0
  
219              max_offenses = options[option_key].to_i
  
220              if detected_offenses > max_offenses
  
221                puts "Metric #{colorize(metric, :green)} detected offenses #{colorize(detected_offenses, :yellow)} is more then #{colorize(max_offenses, :yellow)}, must improve your code quality or set a lower #{colorize(option_key, :black, :white)}"
  
222                audit_failures << {metric: metric, detected_offenses: detected_offenses, max_offenses: max_offenses}
  
223              end
  
224            end
  
225          end
  
226          audit_faild "#{audit_failures.size} of #{selected_metrics.size} metrics audit failed" if audit_failures.any?
  
227        end
  
228      end
  
229    end

248  task :helpers do
  
249    def run_audit(task, title, &block)
  
250      task_name = task.name.split(":").last
  
251      @audit_tasks ||= {}
  
252      @audit_tasks[task_name] ||= {
  
253        report_path: "",
  
254        failure: "",
  
255      }
  
256      puts "## #{title}"
  
257      puts "", "```"
  
258      exc = nil
  
259      begin
  
260        realtime(&block)
  
261      rescue SystemExit => exc
  
262        # audit faild
  
263        @audit_tasks[task_name][:failure] = exc.message.gsub(/(\e\[\d+m)/, "")
  
264      ensure
  
265        # get @report_path set in each audit task
  
266        @audit_tasks[task_name][:report_path] = @report_path.sub("tmp/code_quality/", "")
  
267      end
  
268      puts "```", ""
  
269      raise exc if exc
  
270    end

272    def realtime(&block)
  
273      require 'benchmark'
  
274      realtime = Benchmark.realtime do
  
275        block.call
  
276      end.round
  
277      process_time = humanize_secs(realtime)
  
278      puts "[ #{process_time} ]"
  
279    end

282    # => 1m

285    def humanize_secs(secs)
  
286      [[60, :s], [60, :m], [24, :h], [1000, :d]].map{ |count, name|
  
287        if secs > 0
  
288          secs, n = secs.divmod(count)
  
289          "#{n.to_i}#{name}"
  
290        end
  
291      }.compact.reverse.join(' ').chomp(' 0s')
  
292    end

303    # e.g.: options_from_env(:a, :b) => {:a => ..., :b => ... }
  
304    def options_from_env(*keys)
  
305      # ENV.to_h.slice(*keys.map(&:to_s)).symbolize_keys! # using ActiveSupport
  
306      ENV.to_h.inject({}) { |opts, (k, v)| keys.include?(k.to_sym) ? opts.merge({k.to_sym => v}) : opts }
  
307    end

309    # set text color, background color using ANSI escape sequences, e.g.:
  
310    #   colors = %w(black red green yellow blue pink cyan white default)
  
311    #   colors.each { |color| puts colorize(color, color) }
  
312    #   colors.each { |color| puts colorize(color, :green, color) }
  
313    def colorize(text, color = "default", bg = "default")
  
314      colors = %w(black red green yellow blue pink cyan white default)
  
315      fgcode = 30; bgcode = 40
  
316      tpl = "\e[%{code}m%{text}\e[0m"
  
317      cov = lambda { |txt, col, cod| tpl % {text: txt, code: (cod+colors.index(col.to_s))} }
  
318      ansi = cov.call(text, color, fgcode)
  
319      ansi = cov.call(ansi, bg, bgcode) if bg.to_s != "default"
  
320      ansi
  
321    end
